 f b n t e

# i: current number
# j: rest of numbers
! loadas i | loadas j <- vars a b c d e f
loadas dec code mod_dec6

# f: 1 to 3 (includes 3)
# b: 1 to 5 (includes 5)
set 1 f b

dec:@readln j
dec:@inc i

dec:@while j
    # is not fizz nor buzz
    set 1 n

    ! copy f | sub 3 <- t
    ifelse t +e
    else t -e
        print 'F 'I 'Z 'Z
        clear n f
    endifelse t -e

    ! copy b | sub 5 <- t
    ifelse t +e
    else t -e
        print 'B 'U 'Z 'Z
        clear n b
    endifelse t -e

    if n
        dec:@write i
    endif n

    print 10

    inc b f

    dec:@inc i
    dec:@dec j
dec:@endwhile j

end
