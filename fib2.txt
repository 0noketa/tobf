
loadas dec6 code mod_dec6
loadas x vars a b c d e f
loadas y vars a b c d e f
loadas z vars a b c d e f

dec6:@set 0 0 0 0 0 1 x

loadas dec3 code mod_dec3
loadas n vars a b c

dec3:@readln n

dec3:@while n
    dec6:@writeln x

    dec6:@copy x z
    dec6:@moveadd y x
    dec6:@move z y

    dec3:@dec n
dec3:@endwhile n

end
