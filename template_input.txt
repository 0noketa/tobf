w i e new result

# loads integer to result

set 1 w
while w
    input i
    
    # change here  10 to any charcode
    sub 10  i

    ifelse i e
        clear new

        set 10  w
        while w
            copyadd result  new

            dec w
        endwhile w

        # change hero  38 to 48 minus selection
        sub 38  i

        moveadd i  new
        copy new  result
        inc w
    else i e
        clear w
    endifelse i e
endwhile w


print result

end
