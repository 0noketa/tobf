w i e new

# tail: const
# result: var
# loads integer to result
:readint_until tail result
    clear result
    set 1 w
    while w
        input i
        
        sub tail  i

        ifelse i +e
            clear new

            set 10  w
            while w
                copyadd result  new

                dec w
            endwhile w

            add tail i
            sub 48  i

            moveadd i  new
            copy new  result
            inc w
        else i -e
            clear w
        endifelse i -e
    endwhile w


end
