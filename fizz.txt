i j  f b n t e

# i: current number
# j: rest of numbers
set 1  i

loadas in code mod_input
    in:readint_until 10 j
unload in

# f: 1 to 3 (includes 3)
# b: 1 to 5 (includes 5)
set 1 f b

while j
    # is not fizz nor buzz
    set 1 n

    ! copy f | sub 3 <- t
    ifelse t +e
    else t -e
        print 'F 'I 'Z 'Z
        clear n f
    endifelse t -e

    ! copy b | sub 5 <- t
    ifelse t +e
    else t -e
        print 'B 'U 'Z 'Z
        clear n b
    endifelse t -e

    if n
        loadas out code mod_print
            out:writeint i
        unload out
    endif n

    print 10

    inc b f

    inc i
    dec j
endwhile j

end
