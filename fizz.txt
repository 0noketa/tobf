i j  f b n t e cf ci cz cb cu cln

# i: current number
# j: rest of numbers
set 1  i

# contains it
resb 3
loadas in code mod_input
    in:readint_until 10 j
unload in
resb 14

add 70 cf ci cb cz cu
add 3 ci
sub 4 cb
add 20 cz cu
sub 5 cu
add 10 cln

# f: 1 to 3 (includes 3)
# b: 1 to 5 (includes 5)
set 1 f b

while j
    # is not fizz nor buzz
    set 1 n

    copy f t
    sub 3 t
    ifelse t +e
    else t -e
        print cf ci cz cz
        clear n f
    endifelse t -e

    copy b t
    sub 5 t
    ifelse t +e
    else t -e
        print cb cu cz cz
        clear n b
    endifelse t -e

    if n
        loadas out code mod_print
            out:writeint i
        unload out
    endif n

    print cln

    inc b f

    inc i
    dec j
endwhile j

end
